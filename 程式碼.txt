package com.example.moodshare


import android.os.Bundle
import androidx.activity.ComponentActivity
import androidx.activity.compose.setContent
import androidx.compose.foundation.layout.*
import androidx.compose.foundation.lazy.LazyColumn
import androidx.compose.foundation.lazy.itemsIndexed
import androidx.compose.material3.*
import androidx.compose.runtime.*
import androidx.compose.ui.Modifier
import androidx.compose.ui.unit.dp



data class Mood(

    val nickname:String,

    val emoji:String,

    val content:String,

    val like:Int = 0

)



class MainActivity : ComponentActivity(){


    override fun onCreate(savedInstanceState: Bundle?) {

        super.onCreate(savedInstanceState)


        setContent {

            MoodShareApp()

        }

    }

}




@Composable
fun MoodShareApp(){


    var currentPage by remember {

        mutableStateOf(0)

    }



    var moods by remember {


        mutableStateOf(

            listOf(

                Mood(

                    "匿名星星",

                    "😢",

                    "最近考試壓力很大"

                ),


                Mood(

                    "匿名旅人",

                    "😊",

                    "今天遇到很開心的事情"

                )


            )

        )


    }




    Scaffold(


        bottomBar = {


            NavigationBar{


                NavigationBarItem(

                    selected=currentPage==0,

                    onClick={

                        currentPage=0

                    },

                    icon={Text("🏠")},

                    label={Text("首頁")}

                )




                NavigationBarItem(

                    selected=currentPage==1,

                    onClick={

                        currentPage=1

                    },

                    icon={Text("➕")},

                    label={Text("發布")}

                )




                NavigationBarItem(

                    selected=currentPage==2,

                    onClick={

                        currentPage=2

                    },

                    icon={Text("👤")},

                    label={Text("我的")}

                )


            }


        }


    ){padding->



        Column(

            modifier=Modifier

                .padding(padding)

        ){



            when(currentPage){


                0 -> HomePage(

                    moods=moods,

                    onLike={index ->


                        moods = moods.mapIndexed{

                                i,item ->


                            if(i==index){


                                item.copy(

                                    like=item.like+1

                                )


                            }else{


                                item


                            }


                        }


                    }


                )



                1 -> PostPage(

                    addMood={newMood->


                        moods = moods + newMood


                    }

                )




                2 -> ProfilePage(

                    moods

                )



            }



        }



    }



}







// =====================
// 首頁
// =====================


@Composable
fun HomePage(

    moods:List<Mood>,

    onLike:(Int)->Unit

){


    Column(

        modifier=Modifier

            .fillMaxSize()

            .padding(20.dp)

    ){



        Text(

            "心語匿名",

            style=MaterialTheme.typography.headlineLarge

        )



        Spacer(

            Modifier.height(15.dp)

        )




        LazyColumn{



            itemsIndexed(moods){index,mood ->



                Card(


                    modifier=Modifier

                        .fillMaxWidth()

                        .padding(8.dp)


                ){



                    Column(

                        modifier=Modifier.padding(15.dp)

                    ){



                        Text(

                            mood.nickname

                        )



                        Text(

                            mood.emoji,

                            style=MaterialTheme.typography.headlineLarge

                        )



                        Text(

                            mood.content

                        )



                        Button(


                            onClick={

                                onLike(index)

                            }


                        ){



                            Text(

                                "❤️ ${mood.like}"

                            )


                        }



                    }



                }




            }




        }





    }



}







// =====================
// 發布頁
// =====================


@Composable
fun PostPage(

    addMood:(Mood)->Unit

){



    var text by remember{

        mutableStateOf("")

    }



    var emoji by remember{

        mutableStateOf("😊")

    }




    Column(

        modifier=Modifier

            .fillMaxSize()

            .padding(20.dp)

    ){



        Text(

            "發布心情",

            style=MaterialTheme.typography.headlineLarge

        )



        Spacer(

            Modifier.height(20.dp)

        )



        Row{



            listOf(

                "😊",

                "😢",

                "😡",

                "😰",

                "😴"

            ).forEach{item->



                Button(

                    onClick={

                        emoji=item

                    },

                    modifier=Modifier.padding(2.dp)

                ){



                    Text(item)


                }



            }



        }




        Spacer(

            Modifier.height(15.dp)

        )



        TextField(


            value=text,


            onValueChange={

                text=it

            },


            modifier=Modifier.fillMaxWidth(),


            label={

                Text("輸入你的心情")

            }


        )





        Button(


            onClick={



                if(text.isNotEmpty()){



                    addMood(

                        Mood(

                            "匿名使用者",

                            emoji,

                            text

                        )

                    )


                    text=""


                }



            },


            modifier=Modifier.padding(top=20.dp)


        ){


            Text("發布")


        }




    }



}







// =====================
// 個人紀錄
// =====================


@Composable
fun ProfilePage(

    moods:List<Mood>

){



    Column(

        modifier=Modifier

            .fillMaxSize()

            .padding(20.dp)

    ){



        Text(

            "我的心情紀錄",

            style=MaterialTheme.typography.headlineLarge

        )



        Spacer(

            Modifier.height(20.dp)

        )




        LazyColumn{



            itemsIndexed(moods){_,mood->



                if(mood.nickname=="匿名使用者"){



                    Card(

                        modifier=Modifier

                            .fillMaxWidth()

                            .padding(5.dp)

                    ){


                        Text(

                            "${mood.emoji}  ${mood.content}",


                            modifier=Modifier.padding(15.dp)


                        )


                    }



                }



            }



        }





    }



}
